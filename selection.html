Selection Sort — 視覺化步驟演示

原理： 選擇排序法每一輪（Pass）會掃描未排序的部分，找出最小值，再將它與未排序區域的第一個元素交換位置。 每輪結束後，最小值被擺到正確位置，已排序區域擴大一格。重複 n-1 輪後，所有元素排列完成。

⏱ 時間複雜度：最差 / 平均 / 最佳 均為 O(n²)。每輪僅做 最多 1 次交換，交換次數上限為 n-1 次。

function selectionSort(arr) {
  let n = arr.length
  for (i = 0; i < n-1; i++) { // 第 i 輪
    let minIdx = i // 假設 i 為最小值位置
    for (j = i+1; j < n; j++) { // 尋找更小的值
      if (arr[j] < arr[minIdx]) { // 找到更小的？
        minIdx = j // 更新最小值位置
      }
    }
    if (minIdx !== i) { // 需要交換？
      [arr[i], arr[minIdx]] = [arr[minIdx], arr[i]] // 交換！
    }
  } // arr[i] 已就位
}
